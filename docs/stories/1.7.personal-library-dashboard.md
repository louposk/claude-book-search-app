# Story 1.7: Personal Library Dashboard

## Story
**As a** book enthusiast,
**I want** a dashboard view of my personal library,
**so that** I can easily navigate between my favorites, reading lists, and search history.

3. Dashboard shows summary statistics (total saved books, list counts, recent activity)

5. Responsive design maintains usability across devices

**Main Dashboard Structure** [Source: Component patterns]:
```typescript
// Location: src/app/components/dashboard/dashboard.ts
@Component({
  selector: 'app-dashboard',
  standalone: true,
  imports: [CommonModule, RouterModule, BookListComponent],
  templateUrl: './dashboard.html',
  styleUrl: './dashboard.css'
})
export class DashboardComponent implements OnInit {
  dashboardData: DashboardData = {
    favorites: { count: 0, recentBooks: [] },
    readingLists: { lists: [], totalBooks: 0 },
    searchHistory: { recentSearches: [], totalSearches: 0 },
    statistics: { totalBooks: 0, listsCount: 0, favoriteBooks: 0 }
  };

  loading: boolean = true;
  error: string | null = null;
  recentActivity: ActivityItem[] = [];

  constructor(
    private userDataService: UserDataService,
    private router: Router
  ) {}

  ngOnInit() {
    this.loadDashboardData();
  }

  async loadDashboardData() {
    this.loading = true;
    try {
      const [favorites, readingLists, searchHistory] = await Promise.all([
        this.userDataService.getFavorites().toPromise(),
        this.userDataService.getReadingLists().toPromise(),
        this.userDataService.getSearchHistory(10).toPromise()
      ]);

      this.dashboardData = {
        favorites: {
          count: favorites?.length || 0,
          recentBooks: favorites?.slice(0, 6).map(f => f.book).filter(Boolean) || []
        },
        readingLists: {
          lists: readingLists || [],
          totalBooks: readingLists?.reduce((sum, list) => sum + (list.bookCount || 0), 0) || 0
        },
        searchHistory: {
          recentSearches: searchHistory?.slice(0, 5) || [],
          totalSearches: searchHistory?.length || 0
        },
        statistics: {
          totalBooks: this.calculateTotalUniqueBooks(favorites, readingLists),
          listsCount: readingLists?.length || 0,
          favoriteBooks: favorites?.length || 0
        }
      };

      this.generateRecentActivity();
      this.loading = false;
    } catch (error) {
      this.error = 'Failed to load dashboard data';
      this.loading = false;
      console.error('Dashboard loading error:', error);
    }
  }

  navigateToSection(section: string) {
    this.router.navigate([`/${section}`]);
  }

  performQuickSearch(query: string, searchType: string) {
    this.router.navigate(['/'], {
      queryParams: { q: query, type: searchType }
    });
  }

  createNewReadingList() {
    this.router.navigate(['/reading-lists'], {
      queryParams: { action: 'create' }
    });
  }

  private calculateTotalUniqueBooks(favorites: any[], readingLists: any[]): number {
    const uniqueBookIds = new Set<string>();

    favorites?.forEach(fav => {
      if (fav.bookId) uniqueBookIds.add(fav.bookId);
    });

    readingLists?.forEach(list => {
      list.books?.forEach((book: any) => {
        if (book.id) uniqueBookIds.add(book.id);
      });
    });

    return uniqueBookIds.size;
  }

  private generateRecentActivity() {
    // Combine recent favorites, list additions, and searches into activity timeline
    this.recentActivity = [
      // Implementation for activity timeline generation
    ];
  }
}

interface DashboardData {
  favorites: {
    count: number;
    recentBooks: Book[];
  };
  readingLists: {
    lists: ReadingList[];
    totalBooks: number;
  };
  searchHistory: {
    recentSearches: SearchHistory[];
    totalSearches: number;
  };
  statistics: {
    totalBooks: number;
    listsCount: number;
    favoriteBooks: number;
  };
}

interface ActivityItem {
  type: 'favorite' | 'list' | 'search';
  description: string;
  timestamp: string;
  data?: any;
}
```
