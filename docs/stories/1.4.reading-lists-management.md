## Story
**As a** book enthusiast,
**I want** to create and manage custom reading lists,

5. Books can be moved between lists and removed from lists entirely

**ReadingListsComponent** [Source: FavoritesComponent pattern]:

// Location: src/app/components/reading-lists/reading-lists.ts
@Component({
  selector: 'app-reading-lists',
  standalone: true,
  imports: [CommonModule, FormsModule, RouterModule],
  templateUrl: './reading-lists.html',
  styleUrl: './reading-lists.css'
})
export class ReadingListsComponent implements OnInit {
  readingLists: ReadingList[] = [];
  loading: boolean = false;
  error: string | null = null;
  showCreateForm: boolean = false;
  editingList: ReadingList | null = null;

  newListForm = {
    name: '',
    description: ''
  };

  constructor(private userDataService: UserDataService) {}

  ngOnInit() {
    this.loadReadingLists();
    this.initializeDefaultLists();
  }

  loadReadingLists() {
    this.loading = true;
    this.userDataService.getReadingLists().subscribe({
      next: (lists) => {
        this.readingLists = lists;
        this.loading = false;
      },
      error: (error) => {
        this.error = 'Failed to load reading lists';
        this.loading = false;
      }
    });
  }

  createList() {
    if (this.newListForm.name.trim()) {
      this.userDataService.createReadingList(this.newListForm).subscribe({
        next: (list) => {
          this.readingLists.push(list);
          this.resetCreateForm();
        },
        error: (error) => console.error('Failed to create list:', error)
      });
    }
  }

  deleteList(listId: number) {
    if (confirm('Are you sure you want to delete this reading list?')) {
      this.userDataService.deleteReadingList(listId).subscribe({
        next: () => {
          this.readingLists = this.readingLists.filter(list => list.id !== listId);
        },
        error: (error) => console.error('Failed to delete list:', error)
      });
    }
  }
}
```

@Component({

export class ReadingListDetailComponent implements OnInit {

  readingList: ReadingListWithBooks | null = null;
  loading: boolean = false;
  error: string | null = null;

  constructor(
    private userDataService: UserDataService,
    private route: ActivatedRoute
  ) {}

  ngOnInit() {
    this.route.params.subscribe(params => {
      this.listId = +params['id'];
      this.loadReadingListDetails();
    });
  }

  loadReadingListDetails() {
    this.loading = true;
    this.userDataService.getReadingListWithBooks(this.listId).subscribe({
      next: (list) => {
        this.readingList = list;
        this.loading = false;
      },
      error: (error) => {
        this.error = 'Failed to load reading list';
        this.loading = false;
      }
    });
  }

  removeBookFromList(bookId: string) {
    this.userDataService.removeBookFromList(this.listId, bookId).subscribe({
      next: () => {
        if (this.readingList) {
          this.readingList.books = this.readingList.books.filter(book => book.id !== bookId);
        }
      },
      error: (error) => console.error('Failed to remove book from list:', error)
    });
  }
}
```

### Default Lists Initialization

**Default Lists Service Method** [Source: UserDataService extension]:
```typescript
// Add to UserDataService
private defaultListsInitialized = false;

initializeDefaultLists(): Observable<ReadingList[]> {
  if (this.defaultListsInitialized) {
    return this.getReadingLists();
  }

  const defaultLists = [
    { name: 'To Read', description: 'Books I want to read' },
    { name: 'Currently Reading', description: 'Books I am currently reading' },
    { name: 'Completed', description: 'Books I have finished reading' }
  ];

  return forkJoin(
    defaultLists.map(list => this.createReadingList(list))
  ).pipe(
    tap(() => this.defaultListsInitialized = true),
    catchError(error => {
      console.error('Failed to initialize default lists:', error);
      return this.getReadingLists(); // Fallback to existing lists
    })
  );
}
```
