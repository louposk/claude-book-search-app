# Story 1.5: Search History Tracking

## Status

## Story
**As a** book enthusiast,
**I want** to see my recent search queries,
**so that** I can quickly repeat previous searches and track my book discovery patterns.

## Acceptance Criteria
1. All search queries automatically saved to search history with timestamps
2. Search history interface displays recent searches with quick re-run capability
3. Search history limited to reasonable number (50-100 recent searches) with automatic cleanup
4. History includes search type (title, author, ISBN, general) for context
5. Option to clear search history or remove individual entries

constructor(
  private http: HttpClient,
  private userDataService: UserDataService  // New injection
) { }

private trackSearchHistory(query: string, searchType: string, resultsCount: number) {
  const searchHistory: SearchHistory = {
    query,
    searchType: searchType as 'title' | 'author' | 'isbn' | 'general',
    resultsCount,
    createdAt: new Date().toISOString()
  };

  // Track asynchronously without blocking search results
  this.userDataService.addSearchHistory(searchHistory).subscribe({
    error: (error) => console.error('Failed to track search history:', error)
  });
}

// Enhance existing search methods

### SearchHistoryComponent Specifications

**Component Structure** [Source: FavoritesComponent pattern]:
```typescript
// Location: src/app/components/search-history/search-history.ts
@Component({
  selector: 'app-search-history',
  standalone: true,
  imports: [CommonModule, FormsModule],
  templateUrl: './search-history.html',
  styleUrl: './search-history.css'
})
export class SearchHistoryComponent implements OnInit {
  searchHistory: SearchHistory[] = [];
  loading: boolean = false;
  error: string | null = null;
  historyLimit: number = 50;

  constructor(
    private userDataService: UserDataService,
    private bookService: BookService,
    private router: Router
  ) {}

  ngOnInit() {
    this.loadSearchHistory();
  }

  loadSearchHistory() {
    this.loading = true;
    this.userDataService.getSearchHistory().subscribe({
      next: (history) => {
        this.searchHistory = history.slice(0, this.historyLimit);
        this.loading = false;
      },
      error: (error) => {
        this.error = 'Failed to load search history';
        this.loading = false;
      }
    });
  }

  rerunSearch(historyItem: SearchHistory) {
    // Navigate back to search with pre-filled query and type
    this.router.navigate(['/'], {
      queryParams: {
        q: historyItem.query,
        type: historyItem.searchType
      }
    });
  }

  removeHistoryItem(itemId: number) {
    this.userDataService.removeSearchHistoryItem(itemId).subscribe({
      next: () => {
        this.searchHistory = this.searchHistory.filter(item => item.id !== itemId);
      },
      error: (error) => console.error('Failed to remove history item:', error)
    });
  }

  clearAllHistory() {
    if (confirm('Are you sure you want to clear all search history?')) {
      this.userDataService.clearSearchHistory().subscribe({
        next: () => {
          this.searchHistory = [];
        },
        error: (error) => console.error('Failed to clear history:', error)
      });
    }
  }

  getSearchTypeDisplay(type: string): string {
    const typeMap = {
      'title': 'Title',
      'author': 'Author',
      'isbn': 'ISBN',
      'general': 'General'
    };
    return typeMap[type] || type;
  }

  formatTimestamp(timestamp: string): string {
    return new Date(timestamp).toLocaleDateString('en-US', {
      month: 'short',
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    });
  }
}
```

### UserDataService Enhancement

**Search History Methods** [Source: Story 1.2 service patterns]:
```typescript
// Add to existing UserDataService
getSearchHistory(limit: number = 100): Observable<SearchHistory[]> {
  return this.http.get<SearchHistory[]>(`${this.baseUrl}/search-history`, {
    params: { limit: limit.toString() }
  }).pipe(
    catchError(error => {
      console.error('Failed to get search history:', error);
      return of([]);
    })
  );
}

addSearchHistory(searchHistory: SearchHistory): Observable<SearchHistory> {
  return this.http.post<SearchHistory>(`${this.baseUrl}/search-history`, searchHistory)
    .pipe(
      catchError(error => {
        console.error('Failed to add search history:', error);
        return of(searchHistory); // Return original for graceful degradation
      })
    );
}

removeSearchHistoryItem(itemId: number): Observable<void> {
  return this.http.delete<void>(`${this.baseUrl}/search-history/${itemId}`)
    .pipe(
      catchError(error => {
        console.error('Failed to remove search history item:', error);
        return of();
      })
    );
}

clearSearchHistory(): Observable<void> {
  return this.http.delete<void>(`${this.baseUrl}/search-history`)
    .pipe(
      catchError(error => {
        console.error('Failed to clear search history:', error);
        return of();
      })
    );
}
```
