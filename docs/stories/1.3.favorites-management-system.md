```typescript
// Add to existing component
@Input() showFavorites: boolean = true;  // Allow disabling favorites in some contexts
@Output() favoriteToggled = new EventEmitter<{book: Book, isFavorite: boolean}>();

favoriteBooks: Set<string> = new Set();  // Track favorite status by book ID

constructor(private userDataService: UserDataService) {}

ngOnInit() {
  this.loadFavoriteStatus();
}

isFavorite(bookId: string): boolean {
  return this.favoriteBooks.has(bookId);
}

toggleFavorite(book: Book) {
  const isFavorite = this.isFavorite(book.id);
  if (isFavorite) {
    this.removeFavorite(book);
  } else {
    this.addFavorite(book);
  }
}

private addFavorite(book: Book) {
  this.userDataService.addFavorite(book.id).subscribe({
    next: () => {
      this.favoriteBooks.add(book.id);
      this.favoriteToggled.emit({book, isFavorite: true});
    },
    error: (error) => console.error('Failed to add favorite:', error)
  });
}
```

### New Component Specifications

**FavoritesComponent** [Source: BookListComponent pattern]:
```typescript
// Location: src/app/components/favorites/favorites.ts
@Component({
  selector: 'app-favorites',
  standalone: true,
  imports: [CommonModule, BookListComponent],
  templateUrl: './favorites.html',
  styleUrl: './favorites.css'
})
export class FavoritesComponent implements OnInit {
  favoriteBooks: Book[] = [];
  loading: boolean = false;
  error: string | null = null;

  constructor(private userDataService: UserDataService) {}

  ngOnInit() {
    this.loadFavorites();
  }

  loadFavorites() {
    this.loading = true;
    this.userDataService.getFavorites().subscribe({
      next: (favorites) => {
        this.favoriteBooks = favorites.map(f => f.book).filter(Boolean);
        this.loading = false;
      },
      error: (error) => {
        this.error = 'Failed to load favorites';
        this.loading = false;
      }
    });
  }

  onFavoriteToggled(event: {book: Book, isFavorite: boolean}) {
    if (!event.isFavorite) {
      // Remove from favorites list when unfavorited
      this.favoriteBooks = this.favoriteBooks.filter(book => book.id !== event.book.id);
    }
  }
}
```
