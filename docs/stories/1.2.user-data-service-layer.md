### Service Implementation Specifications

**File Location** [Source: Current project structure]:
- Create: `src/app/services/user-data.service.ts`
- Import pattern: `import { UserDataService } from '../services/user-data.service';`

**Service Structure** [Source: BookService pattern]:
```typescript
@Injectable({
  providedIn: 'root'
})
export class UserDataService {
  private readonly baseUrl = 'http://localhost:3000/api';  // From environment

  constructor(private http: HttpClient) { }

  // Favorites methods
  getFavorites(): Observable<FavoriteResponse[]>
  addFavorite(bookId: string): Observable<Favorite>
  removeFavorite(bookId: string): Observable<void>

  // Reading Lists methods
  getReadingLists(): Observable<ReadingList[]>
  createReadingList(list: ReadingList): Observable<ReadingList>
  updateReadingList(id: number, list: ReadingList): Observable<ReadingList>
  deleteReadingList(id: number): Observable<void>
  addBookToList(listId: number, bookId: string): Observable<ReadingListBook>
  removeBookFromList(listId: number, bookId: string): Observable<void>

  // Search History methods
  getSearchHistory(): Observable<SearchHistory[]>
  addSearchHistory(query: SearchHistory): Observable<SearchHistory>
  clearSearchHistory(): Observable<void>
}
```
